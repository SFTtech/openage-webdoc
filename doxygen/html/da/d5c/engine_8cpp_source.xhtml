241 bool Engine::on_resize(coord::viewport_delta new_size) {
242  // TODO: move all this to the renderer!
243 
244  log::log(MSG(dbg) << "engine window resize to "
245  << new_size.x << "x" << new_size.y);
246 
247  // update engine window size
248  this->coord.viewport_size = new_size;
249 
250  // update camgame window position, set it to center.
251  this->coord.camgame_viewport = coord::viewport{0, 0} + (this->coord.viewport_size / 2);
252 
253  // update camhud window position, set to lower left corner
254  this->coord.camhud_viewport = {0, coord::pixel_t{this->coord.viewport_size.y}};
255 
256  // reset previous projection matrix
257  glMatrixMode(GL_PROJECTION);
258  glLoadIdentity();
259 
260  // update OpenGL viewport: the rendering area
261  glViewport(0, 0, this->coord.viewport_size.x, this->coord.viewport_size.y);
262 
263  // set orthographic projection: left, right, bottom, top, near_val, far_val
264  glOrtho(0, this->coord.viewport_size.x, 0, this->coord.viewport_size.y, 9001, -1);
265 
266  // reset the modelview matrix
267  glMatrixMode(GL_MODELVIEW);
268  glLoadIdentity();
269 
270  return true;
271 }

313 void Engine::loop() {
314  SDL_Event event;
315  util::Timer cap_timer;
316 
317  while (this->running) {
318  this->profiler.start_frame_measure();
319  this->fps_counter.frame();
320  cap_timer.reset(false);
321 
322  this->job_manager.execute_callbacks();
323 
324  this->profiler.start_measure("events", {1.0, 0.0, 0.0});
325  // top level input handling
326  while (SDL_PollEvent(&event)) {
327  switch (event.type) {
328 
329  case SDL_QUIT:
330  this->stop();
331  break;
332 
333  case SDL_WINDOWEVENT: {
334  if (event.window.event == SDL_WINDOWEVENT_RESIZED) {
335  coord::viewport_delta new_size{event.window.data1, event.window.data2};
336 
337  // call additional handlers for the resize event
338  for (auto &handler : on_resize_handler) {
339  if (!handler->on_resize(new_size)) {
340  break;
341  }
342  }
343  }
344  break;
345  }
346 
347  default:
348  for (auto &action : this->on_input_event) {
349  if (false == action->on_input(&event)) {
350  break;
351  }
352  }
353  } // switch event
354  }
355 
356  // here, call to Qt and process all the gui events.
357  this->gui->process_events();
358 
359  if (this->game) {
360  // read camera movement input keys and/or cursor location,
361  // and move camera accordingly.
362 
363  // camera movement speed, in pixels per millisecond
364  // one pixel per millisecond equals 14.3 tiles/second
365  float mov_x = 0.0, mov_y = 0.0, cam_movement_speed_keyboard = 0.5;
366  Uint32 win_flags = SDL_GetWindowFlags(this->window);
367  bool inp_focus = win_flags & SDL_WINDOW_INPUT_FOCUS;
368 
369  input::InputManager &input = this->get_input_manager();
370  using Edge = input::InputManager::Edge;
371 
372  if (inp_focus and (input.is_down(SDLK_LEFT) or
373  input.is_mouse_at_edge(Edge::LEFT, coord.viewport_size.x))) {
374  mov_x = -cam_movement_speed_keyboard;
375  }
376  if (inp_focus and (input.is_down(SDLK_RIGHT) or
377  input.is_mouse_at_edge(Edge::RIGHT, coord.viewport_size.x))) {
378  mov_x = cam_movement_speed_keyboard;
379  }
380  if (inp_focus and (input.is_down(SDLK_DOWN) or
381  input.is_mouse_at_edge(Edge::DOWN, coord.viewport_size.y))) {
382  mov_y = cam_movement_speed_keyboard;
383  }
384  if (inp_focus and (input.is_down(SDLK_UP) or
385  input.is_mouse_at_edge(Edge::UP, coord.viewport_size.y))) {
386  mov_y = -cam_movement_speed_keyboard;
387  }
388 
389  // perform camera movement
390  this->move_phys_camera(
391  mov_x, mov_y,
392  static_cast<float>(this->lastframe_duration_nsec()) / 1e6);
393 
394  // update the currently running game
395  this->game->update(this->lastframe_duration_nsec());
396  }
397  this->profiler.end_measure("events");
398 
399  // call engine tick callback methods
400  for (auto &action : this->on_engine_tick) {
401  if (false == action->on_tick()) {
402  break;
403  }
404  }
405 
406  this->profiler.start_measure("rendering", {0.0, 1.0, 0.0});
407 
408  // clear the framebuffer to black
409  // in the future, we might disable it for lazy drawing
410  glClearColor(0.0, 0.0, 0.0, 0.0);
411  glClear(GL_COLOR_BUFFER_BIT | GL_STENCIL_BUFFER_BIT);
412 
413  // invoke all game drawing handlers
414  for (auto &action : this->on_drawgame) {
415  if (false == action->on_draw()) {
416  break;
417  }
418  }
419 
420  util::gl_check_error();
421 
422  // draw the fps overlay
423  if (this->drawing_debug_overlay.value) {
424  this->draw_debug_overlay();
425  }
426 
427  // invoke all hud drawing callback methods
428  if (this->drawing_huds.value) {
429  for (auto &action : this->on_drawhud) {
430  if (false == action->on_drawhud()) {
431  break;
432  }
433  }
434  }
435 
436  this->text_renderer->render();
437 
438  util::gl_check_error();
439 
440  this->profiler.end_measure("rendering");
441 
442  this->profiler.start_measure("idle", {0.0, 0.0, 1.0});
443 
444  // the rendering is done
445  // swap the drawing buffers to actually show the frame
446  SDL_GL_SwapWindow(window);
447 
448  if (this->ns_per_frame != 0) {
449  uint64_t ns_for_current_frame = cap_timer.getval();
450  if (ns_for_current_frame < this->ns_per_frame) {
451  SDL_Delay((this->ns_per_frame - ns_for_current_frame) / 1e6);
452  }
453  }
454 
455  this->profiler.end_measure("idle");
456 
457  this->profiler.end_frame_measure();
458  }
459 }
