129 def verbosity_to_level(verbosity):
130  """
131  Translates an integer verbosity to a log level.
132  """
133  levels = [logging.getLevelName("MIN"),
134  logging.getLevelName("SPAM"),
135  logging.DEBUG,

138  logging.ERROR,
