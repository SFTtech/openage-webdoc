32  def get_direntries(self, parts=None, create=False):
33  """
34  Fetches the fileentries, subdirentries tuple for the given dir.
35 
36  If create == False, raises FileNotFoundError if the directory doesn't

39  Helper method for internal use.
40  """

47  if create:
