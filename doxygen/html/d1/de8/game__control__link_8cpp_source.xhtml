147 void ActionModeLink::on_selection_changed(const UnitSelection *unit_selection, const Player *player) {
148  this->selection = unit_selection;
149 
150  if (this->selection->get_units_count() == 1) {
151  auto &ref = this->selection->get_first_unit();
152  if (ref.is_valid()) {
153  Unit *u = ref.get();
154 
155  this->selection_name = QString::fromStdString(u->unit_type->name());
156  // the icons are split into two sprites
157  if (u->unit_type->unit_class == gamedata::unit_classes::BUILDING) {
158  this->selection_icon = QString::fromStdString("50706.slp.png." + std::to_string(u->unit_type->icon));
159  } else {
160  this->selection_icon = QString::fromStdString("50730.slp.png." + std::to_string(u->unit_type->icon));
161  }
162  this->selection_type = QString::fromStdString("(type: " + std::to_string(u->unit_type->id()) + " " + u->top()->name() + ")");
163 
164  if (u->has_attribute(attr_type::owner)) {
165  auto &own_attr = u->get_attribute<attr_type::owner>();
166  if (own_attr.player.civ->civ_id != 0) { // not gaia
167  this->selection_owner = QString::fromStdString(
168  own_attr.player.name + "\n" + own_attr.player.civ->civ_name + "\n" +
169  (!player || *player == own_attr.player ? ""
170  : player->is_ally(own_attr.player) ? "Ally" : "Enemy")
171  );
172  // TODO find the team status of the player
173  } else {
174  this->selection_owner = QString::fromStdString(" ");
175  }
176  }
177 
178  if (u->has_attribute(attr_type::hitpoints) && u->has_attribute(attr_type::damaged)) {
179  auto &hp = u->get_attribute<attr_type::hitpoints>();
180  auto &dm = u->get_attribute<attr_type::damaged>();
181  // TODO replace ascii health bar with real one
182  if (hp.hp >= 200) {
183  this->selection_hp = QString::fromStdString(progress(dm.hp*1.0f/hp.hp, 8)+" "+std::to_string(dm.hp)+"/"+std::to_string(hp.hp));
184  } else {
185  this->selection_hp = QString::fromStdString(progress(dm.hp*1.0f/hp.hp, 4)+" "+std::to_string(dm.hp)+"/"+std::to_string(hp.hp));
186  }
187  } else {
188  this->selection_hp = QString::fromStdString(" ");
189  }
190 
191  std::string lines;
192  if (u->has_attribute(attr_type::resource)) {
193  auto &res_attr = u->get_attribute<attr_type::resource>();
194  lines += std::to_string((int) res_attr.amount)+" "+std::to_string(res_attr.resource_type) + "\n";
195  }
196  if (u->has_attribute(attr_type::building)) {
197  auto &build_attr = u->get_attribute<attr_type::building>();
198  if (build_attr.completed < 1) {
199  lines += "Building: " + progress(build_attr.completed, 16)+ " "+std::to_string((int) (100 * build_attr.completed)) + "%\n";
200  }
201  }
202  if (u->has_attribute(attr_type::garrison)) {
203  auto &garrison_attr = u->get_attribute<attr_type::garrison>();
204  if (garrison_attr.content.size() > 0) {
205  lines += "Garrison: "+std::to_string(garrison_attr.content.size()) + " units\n";
206  }
207  }
208  lines += "\n";
209  if (u->has_attribute(attr_type::population)) {
210  auto &population_attr = u->get_attribute<attr_type::population>();
211  if (population_attr.demand > 1) {
212  lines += "Population demand: "+std::to_string(population_attr.demand) + " units\n";
213  }
214  if (population_attr.capacity > 0) {
215  lines += "Population capacity: "+std::to_string(population_attr.capacity) + " units\n";
216  }
217  }
218  this->selection_attrs = QString::fromStdString(lines);
219 
220  }
221 
222  } else if (this->selection->get_units_count() > 1) {
223 
224  this->selection_name = QString::fromStdString(
225  std::to_string(this->selection->get_units_count()) + " units");
226  }
227 
228 
229  emit this->selection_changed();
230 }
