128 void UnitSelection::add_unit(const Player &player, Unit *u, bool append) {
129  // Only select resources and units with hitpoints > 0
130  if (u->has_attribute(attr_type::resource) ||
131  (u->has_attribute(attr_type::damaged) && u->get_attribute<attr_type::damaged>().hp > 0)) {
132 
133  selection_type_t unit_type = get_unit_selection_type(player, u);
134  int unit_type_i = static_cast<int>(unit_type);
135  int selection_type_i = static_cast<int>(this->selection_type);
136 
137  if (unit_type_i > selection_type_i) {
138  // Don't select this unit as it has too low priority
139  return;
140  }
141 
142  if (unit_type_i < selection_type_i) {
143  // Upgrade selection to a higher priority selection
144  this->clear();
145  this->selection_type = unit_type;
146  }
147 
148  // Can't select multiple enemies at once
149  if (not (unit_type == selection_type_t::own_units ||
150  (unit_type == selection_type_t::own_buildings && append))) {
151 
152  this->clear();
153  this->selection_type = unit_type; // Clear resets selection_type
154  }
155 
156  // Finally, add the unit to the selection
157  u->selected = true;
158  this->units[u->id] = u->get_ref();
159  }
160 }

238 void UnitSelection::select_space(const Player &player, Terrain *terrain,
239  coord::camgame point0, coord::camgame point1, bool append) {
240  if (!append) {
241  this->clear();
242  }
243 
244  coord::camgame min{std::min(point0.x, point1.x), std::min(point0.y, point1.y)};
245  coord::camgame max{std::max(point0.x, point1.x), std::max(point0.y, point1.y)};
246 
247  // look at each tile in the range and find all units
248  for (coord::tile check_pos : tiles_in_range(point0, point1, this->engine->coord)) {
249  TileContent *tc = terrain->get_data(check_pos);
250  if (tc) {
251  // find objects within selection box
252  for (auto unit_location : tc->obj) {
253  coord::camgame pos = unit_location->pos.draw.to_camgame(this->engine->coord);
254 
255  if ((pos.x > min.x and pos.x < max.x) and
256  (pos.y > min.y and pos.y < max.y)) {
257  this->add_unit(player, &unit_location->unit, append);
258  }
259  }
260  }
261  }

287 std::vector<coord::tile> tiles_in_range(coord::camgame p1, coord::camgame p2, const coord::CoordManager &coord) {
288  // the remaining corners
289  coord::camgame p3 = coord::camgame{p1.x, p2.y};
290  coord::camgame p4 = coord::camgame{p2.x, p1.y};
291  coord::camgame pts[4] {p1, p2, p3, p4};
292 
293  // find the range of tiles covered
294  coord::tile t1 = pts[0].to_tile(coord);
295  coord::tile min = t1;
296  coord::tile max = t1;
297 
298  for (unsigned i = 1; i < 4; ++i) {
299  coord::tile t = pts[i].to_tile(coord);
300  min.ne = std::min(min.ne, t.ne);
301  min.se = std::min(min.se, t.se);
302  max.ne = std::max(max.ne, t.ne);
303  max.se = std::max(max.se, t.se);
304  }
305 
306  // find all units in the boxed region
307  std::vector<coord::tile> tiles;
308  coord::tile check_pos = min;
309  while (check_pos.ne <= max.ne) {
310  while (check_pos.se <= max.se) {
311  tiles.push_back(check_pos);
312  check_pos.se += 1;
313  }
314  check_pos.se = min.se;
315  check_pos.ne += 1;
316  }
317  return tiles;
318 }
